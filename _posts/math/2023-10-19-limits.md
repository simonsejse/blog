$$
\begin{align*}
         \int u\, dv &= uv-\int v \, du \\
         \int x \, e^{-x} \, dx &= -xe^{-x}-\int -e^{-x}dx \\ 
         &= -xe^{-x}-(e^{-x})= -xe^{-x}-e^{-x} \\ 
         &= e^{-x}(-x-1)
\end{align*}
