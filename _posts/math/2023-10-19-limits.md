$$
 \begin{align*}
         &\int_{-\y}^{\y}xf(x) \, dx = \lim_{{a \to -\y}}\int_{a}^{0}{x(x^2 exp(x)) dx} + \lim_{{a \to \y}} \int_{0}^{a}{x(x*exp(-x)) dx}  \\
         &= \left( \left. \lim_{{a \to -\y}} e^x(x^3 - 3x^2 + 6x - 6) \right|_{a}^{0} \right) + \left( \left. \lim_{{a \to \y}} e^{-x}(-x^2 - 2x-2) \right|_{0}^{a} \right) \\
         &= -6 + 2 = -4
  \end{align*}
$$
